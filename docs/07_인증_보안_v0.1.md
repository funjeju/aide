## 1. 인증 흐름

### 1.1 사용자 인증 (모바일 앱)

```
[시작]
  ↓
사용자가 로그인 화면에서 [Google] 또는 [Apple] 선택
  ↓
[Google Sign-In] (Android) / Apple Sign-In (iOS)
  ↓
OAuth flow → Firebase Auth ID Token
  ↓
앱이 토큰 보유 (자동 갱신)
  ↓
모든 API 호출에 ID Token 포함 (Callable이 자동 처리)
  ↓
Cloud Functions가 토큰 검증 (request.auth.uid)
  ↓
Firestore Rules가 토큰 검증 (request.auth.uid)
```

### 1.2 어드민 인증 (Next.js 어드민 콘솔)

```
어드민이 어드민 콘솔 접속
  ↓
이메일 + 비밀번호 입력
  ↓
Firebase Auth 인증
  ↓
[2FA 단계]
  TOTP 코드 입력 (Google Authenticator)
  ↓
검증 성공 → Custom Claim 확인 (admin: true)
  ↓
세션 쿠키 발급 (HttpOnly + Secure + SameSite=Strict)
  ↓
어드민 콘솔 진입
  ↓
모든 액션 → admin_logs 자동 기록
```

### 1.3 OAuth Provider 설정

| Provider | iOS | Android | Web (어드민) |
|---|---|---|---|
| Google Sign-In | ✅ | ✅ (메인) | ✅ |
| Apple Sign-In | ✅ (메인) | ❌ | ❌ |
| Email/Password | ❌ | ❌ | ✅ (어드민만) |

### 1.4 어드민 계정 생성 정책
- **셀프 가입 불가** (콘솔 화면에 가입 폼 없음)
- 슈퍼 어드민이 다른 어드민 콘솔에서 직접 추가
- 신규 어드민은 첫 로그인 시 비밀번호 변경 + 2FA 설정 필수

## 2. Firestore Security Rules 풀 버전

### 2.1 전체 규칙

```javascript
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ================================================
    // 헬퍼 함수
    // ================================================

    function isSignedIn() {
      return request.auth != null;
    }

    function isOwner(uid) {
      return isSignedIn() && request.auth.uid == uid;
    }

    function isAdmin() {
      return isSignedIn() && request.auth.token.admin == true;
    }

    function isSuperAdmin() {
      return isAdmin() && request.auth.token.role == 'super_admin';
    }

    function isOwnerOrAdmin(uid) {
      return isOwner(uid) || isAdmin();
    }

    function hasShareAccess(resourceId) {
      return isSignedIn() && exists(
        /databases/$(database)/documents/shares/$(resourceId)/members/$(request.auth.uid)
      );
    }

    function getShareRole(resourceId) {
      return get(
        /databases/$(database)/documents/shares/$(resourceId)/members/$(request.auth.uid)
      ).data.role;
    }

    function hasEditAccess(resourceId) {
      return hasShareAccess(resourceId) &&
             getShareRole(resourceId) in ['owner', 'editor'];
    }

    function hasCommentAccess(resourceId) {
      return hasShareAccess(resourceId) &&
             getShareRole(resourceId) in ['owner', 'editor', 'commenter'];
    }

    function isValidString(field, maxLen) {
      return field is string && field.size() > 0 && field.size() <= maxLen;
    }

    // ================================================
    // 사용자 컬렉션
    // ================================================

    match /users/{uid} {
      allow read: if isOwnerOrAdmin(uid);
      allow create: if isOwner(uid)
                       && request.resource.data.uid == uid
                       && request.resource.data.userTier == 'FREE';
      allow update: if isOwner(uid)
                       && !request.resource.data.diff(resource.data).affectedKeys()
                            .hasAny(['userTier', 'userLimitsOverride', 'tierHistory', 'uid']);
      allow update: if isAdmin();
      allow delete: if false;

      match /preferences/{docId} {
        allow read, write: if isOwnerOrAdmin(uid);
      }

      match /integrations/{docId} {
        allow read: if isOwner(uid);
        allow write: if false;  // OAuth 토큰 노출 방지
      }

      match /locations/{locationId} {
        allow read, write: if isOwner(uid);
      }

      match /habits/{habitId} {
        allow read, write: if isOwner(uid);
      }

      match /insights/{insightId} {
        allow read: if isOwner(uid);
        allow write: if false;
      }

      match /usage/{date} {
        allow read: if isOwner(uid);
        allow write: if false;
      }

      match /dailyReports/{date} {
        allow read: if isOwner(uid);
        allow write: if false;
      }

      match /businessReports/{period}/{date} {
        allow read: if isOwner(uid);
        allow write: if false;
      }

      match /dashboard/{cacheId} {
        allow read: if isOwner(uid);
        allow write: if false;
      }

      match /contextSummary/{summaryId} {
        allow read: if isOwner(uid);
        allow write: if false;
      }
    }

    // ================================================
    // 데이터 흐름 (Raw → Parsed → Draft)
    // ================================================

    match /rawData/{rawId} {
      allow read: if isOwner(resource.data.userId);
      allow create: if isOwner(request.resource.data.userId)
                       && isValidString(request.resource.data.content, 50000);
      allow update: if false;
      allow delete: if isOwner(resource.data.userId);
    }

    match /parsedData/{parsedId} {
      allow read: if isOwner(resource.data.userId);
      allow create: if false;
      allow update: if false;
      allow delete: if isOwner(resource.data.userId);
    }

    match /drafts/{draftId} {
      allow read: if isOwner(resource.data.userId);
      allow create: if false;
      allow update: if isOwner(resource.data.userId)
                       && request.resource.data.userId == resource.data.userId;
      allow delete: if isOwner(resource.data.userId);
    }

    // ================================================
    // 확정 엔티티 (Tasks / Events / Persons)
    // ================================================

    match /tasks/{taskId} {
      allow read: if isOwner(resource.data.userId)
                     || hasShareAccess(taskId);
      allow create: if isOwner(request.resource.data.userId)
                       && isValidString(request.resource.data.title, 200)
                       && request.resource.data.priority in ['URGENT', 'HIGH', 'NORMAL', 'LOW', 'SOMEDAY']
                       && request.resource.data.track in ['LIFE', 'WORK'];
      allow update: if (isOwner(resource.data.userId) || hasEditAccess(taskId))
                       && request.resource.data.userId == resource.data.userId;
      allow delete: if isOwner(resource.data.userId);

      match /comments/{commentId} {
        allow read: if isOwner(get(/databases/$(database)/documents/tasks/$(taskId)).data.userId)
                       || hasShareAccess(taskId);
        allow create: if isSignedIn()
                         && (isOwner(get(/databases/$(database)/documents/tasks/$(taskId)).data.userId)
                             || hasCommentAccess(taskId));
        allow update: if isOwner(resource.data.userId);
        allow delete: if isOwner(resource.data.userId);
      }

      match /activities/{activityId} {
        allow read: if isOwner(get(/databases/$(database)/documents/tasks/$(taskId)).data.userId)
                       || hasShareAccess(taskId);
        allow write: if false;
      }

      match /attachments/{attachmentId} {
        allow read: if isOwner(get(/databases/$(database)/documents/tasks/$(taskId)).data.userId)
                       || hasShareAccess(taskId);
        allow create: if isOwner(get(/databases/$(database)/documents/tasks/$(taskId)).data.userId);
        allow delete: if isOwner(resource.data.userId);
      }
    }

    match /events/{eventId} {
      allow read: if isOwner(resource.data.userId)
                     || hasShareAccess(eventId);
      allow create: if isOwner(request.resource.data.userId)
                       && isValidString(request.resource.data.title, 200);
      allow update: if (isOwner(resource.data.userId) || hasEditAccess(eventId))
                       && request.resource.data.userId == resource.data.userId;
      allow delete: if isOwner(resource.data.userId);
    }

    match /persons/{personId} {
      allow read: if isOwner(resource.data.userId);
      allow create: if isOwner(request.resource.data.userId)
                       && isValidString(request.resource.data.name, 100);
      allow update: if isOwner(resource.data.userId);
      allow delete: if isOwner(resource.data.userId);
    }

    // ================================================
    // 만다라트
    // ================================================

    match /projects/{projectId} {
      allow read: if isOwner(resource.data.userId)
                     || hasShareAccess(projectId);
      allow create: if isOwner(request.resource.data.userId);
      allow update: if (isOwner(resource.data.userId) || hasEditAccess(projectId))
                       && request.resource.data.userId == resource.data.userId;
      allow delete: if isOwner(resource.data.userId);
    }

    match /sections/{sectionId} {
      allow read: if isOwner(resource.data.userId)
                     || hasShareAccess(resource.data.projectId);
      allow create: if isOwner(request.resource.data.userId);
      allow update: if isOwner(resource.data.userId) || hasEditAccess(resource.data.projectId);
      allow delete: if isOwner(resource.data.userId);
    }

    match /subSections/{subSectionId} {
      allow read: if isOwner(resource.data.userId)
                     || hasShareAccess(resource.data.projectId);
      allow create: if isOwner(request.resource.data.userId);
      allow update: if isOwner(resource.data.userId) || hasEditAccess(resource.data.projectId);
      allow delete: if isOwner(resource.data.userId);
    }

    match /mandalartTasks/{mandalartTaskId} {
      allow read: if isOwner(resource.data.userId)
                     || hasShareAccess(resource.data.projectId);
      allow create: if isOwner(request.resource.data.userId);
      allow update: if isOwner(resource.data.userId) || hasEditAccess(resource.data.projectId);
      allow delete: if isOwner(resource.data.userId);
    }

    match /mandalartTemplates/{templateId} {
      allow read: if isSignedIn();
      allow write: if isAdmin();
    }

    // ================================================
    // 감정 / 챗봇
    // ================================================

    match /emotionTags/{tagId} {
      allow read: if isOwner(resource.data.userId);
      allow create: if isOwner(request.resource.data.userId);
      allow update: if isOwner(resource.data.userId);
      allow delete: if isOwner(resource.data.userId);
    }

    match /chats/{chatId} {
      allow read: if isOwner(resource.data.userId);
      allow create: if isOwner(request.resource.data.userId);
      allow update: if isOwner(resource.data.userId);
      allow delete: if isOwner(resource.data.userId);

      match /messages/{messageId} {
        allow read: if isOwner(get(/databases/$(database)/documents/chats/$(chatId)).data.userId);
        allow create: if false;
        allow update: if isOwner(get(/databases/$(database)/documents/chats/$(chatId)).data.userId)
                         && request.resource.data.diff(resource.data).affectedKeys()
                              .hasOnly(['feedback', 'flagged']);
        allow delete: if false;
      }
    }

    // ================================================
    // 반복 Task / 협업 / 결제 / 내보내기
    // ================================================

    match /recurringTasks/{recurringId} {
      allow read: if isOwner(resource.data.userId);
      allow create: if isOwner(request.resource.data.userId);
      allow update: if isOwner(resource.data.userId);
      allow delete: if isOwner(resource.data.userId);
    }

    match /shares/{shareId} {
      allow read: if isOwner(resource.data.ownerId)
                     || hasShareAccess(shareId);
      allow create: if isOwner(request.resource.data.ownerId);
      allow update: if isOwner(resource.data.ownerId);
      allow delete: if isOwner(resource.data.ownerId);

      match /members/{userId} {
        allow read: if isOwner(get(/databases/$(database)/documents/shares/$(shareId)).data.ownerId)
                       || isOwner(userId);
        allow create: if isOwner(get(/databases/$(database)/documents/shares/$(shareId)).data.ownerId);
        allow update: if isOwner(userId)
                         && request.resource.data.diff(resource.data).affectedKeys()
                              .hasOnly(['status', 'acceptedAt', 'lastViewedAt']);
        allow delete: if isOwner(get(/databases/$(database)/documents/shares/$(shareId)).data.ownerId);
      }
    }

    match /publicLinks/{linkId} {
      allow read: if true;  // 공개 링크는 인증 없이도 읽기 가능
      allow create: if isOwner(request.resource.data.createdBy);
      allow update: if isOwner(resource.data.createdBy);
      allow delete: if isOwner(resource.data.createdBy);
    }

    match /subscriptions/{subscriptionId} {
      allow read: if isOwner(resource.data.userId) || isAdmin();
      allow write: if false;
    }

    match /payments/{paymentId} {
      allow read: if isOwner(resource.data.userId) || isAdmin();
      allow write: if false;
    }

    match /exports/{exportId} {
      allow read: if isOwner(resource.data.userId);
      allow create: if isOwner(request.resource.data.userId);
      allow update: if false;
      allow delete: if isOwner(resource.data.userId);
    }

    // ================================================
    // 어드민 시스템
    // ================================================

    match /admin_users/{adminId} {
      allow read: if isAdmin();
      allow write: if isSuperAdmin();
    }

    match /tiers/{tierCode} {
      allow read: if isSignedIn();
      allow write: if isAdmin();
    }

    match /admin_logs/{logId} {
      allow read: if isAdmin();
      allow write: if false;
    }

    match /system_announcements/{id} {
      allow read: if isSignedIn();
      allow write: if isAdmin();
    }

    match /email_templates/{templateId} {
      allow read: if isAdmin();
      allow write: if isAdmin();
    }

    match /ai_prompts/{tier}/versions/{versionId} {
      allow read: if isAdmin();
      allow write: if isAdmin();
    }

    match /system_metrics/{date} {
      allow read: if isAdmin();
      allow write: if false;
    }

    // ================================================
    // 기본 거부 (default deny)
    // ================================================

    match /{document=**} {
      allow read, write: if false;
    }
  }
}
```

### 3.2 한도 우선순위

```
1. UserLimitsOverride (어드민 수동 조정) — users/{uid}.userLimitsOverride
   ↓ null인 경우
2. Tier 기본값 — tiers/{tierCode}.limits
```

### 3.3 한도 체크 흐름

```typescript
// functions/src/services/limits/checkLimit.ts
export async function checkUserLimit(
  uid: string,
  action: LimitAction,
  amount = 1
): Promise<LimitCheckResult> {
  // 1. 사용자 한도 조회
  const userDoc = await db.doc(`users/${uid}`).get();
  const user = userDoc.data();
  const limits = user.userLimitsOverride
    ?? (await db.doc(`tiers/${user.userTier}`).get()).data().limits;

  const limit = mapActionToLimit(action, limits);

  // 2. 무제한이면 통과
  if (limit === -1) {
    return { allowed: true, limit: -1, used: 0, remaining: -1 };
  }

  // 3. 오늘 사용량 조회
  const today = new Date().toISOString().split('T')[0];
  const usageDoc = await db.doc(`users/${uid}/usage/${today}`).get();
  const used = usageDoc.data()?.[mapActionToUsageKey(action)] ?? 0;

  // 4. 비교
  if (used + amount > limit) {
    return {
      allowed: false,
      reason: 'LIMIT_EXCEEDED',
      limit,
      used,
      upgradeUrl: '...',
    };
  }

  return { allowed: true, limit, used, remaining: limit - used };
}
```

### 3.4 동시성 처리 (Race Condition)

**문제**: 사용자가 동시에 여러 요청 → 한도 체크와 증가 사이에 race.

**해결 (트랜잭션)**:
```typescript
await db.runTransaction(async (tx) => {
  const usageDoc = await tx.get(usageRef);
  const used = usageDoc.data()?.[usedKey] ?? 0;
  if (used + amount > limit) {
    throw new HttpsError('resource-exhausted', '한도 초과');
  }
  tx.set(usageRef, { [usedKey]: used + amount }, { merge: true });
});
```

## 4. 어드민 권한 모델

### 4.1 Custom Claims 구조

```typescript
{
  admin: true,
  role: 'super_admin' | 'admin' | 'support',
  permissions?: string[],
}
```

### 4.2 역할별 권한

| 역할 | 권한 |
|---|---|
| `super_admin` | 모든 권한 + 어드민 추가/삭제 |
| `admin` | 사용자 관리, Tier 변경, 콘텐츠 관리, 통계 조회 |
| `support` | 사용자 조회, 한도 임시 부여, 환불 처리 (제한적) |

### 4.3 Custom Claims 부여

```typescript
export const setAdminClaim = onCall(async (request) => {
  if (!request.auth?.token.admin || request.auth.token.role !== 'super_admin') {
    throw new HttpsError('permission-denied', '슈퍼 어드민만 가능');
  }

  const { targetUid, role } = request.data;

  await admin.auth().setCustomUserClaims(targetUid, {
    admin: true,
    role,
  });

  await db.doc(`admin_users/${targetUid}`).set({
    id: targetUid,
    email: (await admin.auth().getUser(targetUid)).email,
    role,
    createdAt: FieldValue.serverTimestamp(),
    createdBy: request.auth.uid,
    isActive: true,
  });

  await db.collection('admin_logs').add({
    adminId: request.auth.uid,
    action: 'admin_claim_granted',
    target: { type: 'admin', id: targetUid },
    after: { role },
    timestamp: FieldValue.serverTimestamp(),
  });

  return { ok: true };
});
```

### 4.4 클라이언트 권한 확인 (Next.js 미들웨어)

```typescript
export async function middleware(req) {
  const session = await getSession();
  const tokenResult = await session.getIdTokenResult();

  if (!tokenResult.claims.admin) {
    return NextResponse.redirect('/login');
  }

  // 역할별 라우트 보호
  if (req.nextUrl.pathname.startsWith('/admin/super') && tokenResult.claims.role !== 'super_admin') {
    return NextResponse.redirect('/admin/forbidden');
  }
}
```

### 4.5 2FA (TOTP)

#### 4.5.1 설정 흐름
```
어드민 첫 로그인
  ↓ "2FA 설정 필요" 안내
  ↓ QR 코드 표시 (TOTP secret)
  ↓ Google Authenticator로 스캔
  ↓ 6자리 코드 입력 (검증)
  ↓ totpSecret을 DB에 저장 (KMS 암호화)
  ↓ 이후 로그인마다 TOTP 코드 요구
```

#### 4.5.2 구현 (speakeasy)

```typescript
import speakeasy from 'speakeasy';

export function generateTotpSecret(email: string) {
  const secret = speakeasy.generateSecret({
    name: `MyApp Admin (${email})`,
    issuer: 'MyApp',
  });
  return {
    secret: secret.base32,
    qrCodeUrl: secret.otpauth_url,
  };
}

export function verifyTotpCode(secret: string, code: string): boolean {
  return speakeasy.totp.verify({
    secret,
    encoding: 'base32',
    token: code,
    window: 1,  // ±30초 허용
  });
}
```

### 5.3 헬퍼

```typescript
import { KeyManagementServiceClient } from '@google-cloud/kms';

const client = new KeyManagementServiceClient();
const KEY_NAME = `projects/${PROJECT_ID}/locations/asia-northeast3/keyRings/app-secrets/cryptoKeys/oauth-tokens`;

export async function encrypt(plaintext: string): Promise<string> {
  const [result] = await client.encrypt({
    name: KEY_NAME,
    plaintext: Buffer.from(plaintext),
  });
  return result.ciphertext.toString('base64');
}

export async function decrypt(ciphertext: string): Promise<string> {
  const [result] = await client.decrypt({
    name: KEY_NAME,
    ciphertext: Buffer.from(ciphertext, 'base64'),
  });
  return result.plaintext.toString();
}
```

## 7. Cloud Storage 보안

### 7.1 Storage Rules

```javascript
rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {

    match /users/{uid}/voice/{audioId} {
      allow read: if request.auth.uid == uid;
      allow write: if request.auth.uid == uid
                      && request.resource.size < 50 * 1024 * 1024
                      && request.resource.contentType.matches('audio/.*');
    }

    match /users/{uid}/attachments/{taskId}/{filename} {
      allow read: if request.auth.uid == uid
                     || hasShareAccess(taskId);
      allow write: if request.auth.uid == uid
                      && request.resource.size < 100 * 1024 * 1024;
    }

    match /users/{uid}/profile/{filename} {
      allow read: if true;  // 프로필은 공개
      allow write: if request.auth.uid == uid
                      && request.resource.size < 5 * 1024 * 1024
                      && request.resource.contentType.matches('image/.*');
    }

    match /exports/{exportId}/{filename} {
      allow read: if request.auth != null;
      allow write: if false;
    }

    match /admin/{path=**} {
      allow read, write: if request.auth.token.admin == true;
    }
  }
}

function hasShareAccess(resourceId) {
  return firestore.exists(
    /databases/(default)/documents/shares/$(resourceId)/members/$(request.auth.uid)
  );
}
```
